The hook_cronapi($hook) allows you to define new cronjobs

Example

// Define custom cron functions
function mymodule_cronapi($hook) {
  $items = array();
  $items['mymodule_cronjob_1'] = array(
    'description' => 'Cron-1 Handler',
    'rules' => array('*/13 * * * *'),
    'configure' => 'admin/configure-modules-settings/xxx',
  );
  $items['mymodule_cronjob_2'] = array(
    'description' => 'Cron-2 Handler',
    'callback' => 'mymodule_somefunction',
    'rules' => array('*/13 * * * *'),
  );
  $items['mymodule_cronjob_3'] = array(
    'description' => 'Cron-3 Handler',
    'enabled' => FALSE,
  );
  return $items;
}

// Custom cron-function
function mymodule_cronjob_1() {
  // Do some stuff ...
}

// Custom cron-function
function mymodule_somefunction() {
  // Do some stuff ...
}

// Custom cron-function
function mymodule_cronjob_3() {
  // Do some stuff ...
}
